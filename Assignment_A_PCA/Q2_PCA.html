def DFFS_Conv(image, kernel, PCA_matrix):
    m, n = kernel.shape
    if (m == n):
        vertical, horizontal = image.shape
        vertical = vertical - m + 1
        horizontal = horizontal - n + 1
        new_image = np.zeros((vertical,horizontal))
        for i in range(vertical):
            for j in range(horizontal):
                normalized_x = (image[i:i+m, j:j+m]-kernel)
                normalized_x_flat = normalized_x.flatten()
                y = np.dot(PCA_matrix, normalized_x_flat)
                normalized_x_sqr = np.power(normalized_x,2)
                y_square = np.power(y,2)
                result = np.sum(normalized_x_sqr) - np.sum(y_square)
                new_image[i][j] = result
    return new_image

def SSD_Conv(image, kernel):
    m, n = kernel.shape
    if (m == n):
        vertical, horizontal = image.shape
        vertical = vertical - m + 1
        horizontal = horizontal - n + 1
        new_image = np.zeros((vertical,horizontal))
        for i in range(vertical):
            for j in range(horizontal):
                normalized_x = (image[i:i+m, j:j+m]-kernel)
                result = np.power(normalized_x,2)
                result = np.sum(result)
                result = math.sqrt(result)
                new_image[i][j] = result
                #new_image[i][j] = np.sum(y_square)
    return new_image
